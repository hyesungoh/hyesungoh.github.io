#### 대입
{% highlight python %}
   # if condition : function or value
   if 5 > 3 : num = 5
   # num = 5
{% endhighlight %}

#### 사용
{% highlight python %}
   if 5 > 3 : print("5는 3보다 큽니다")
   # prints "5는 3보다 큽니다"
{% endhighlight %}
